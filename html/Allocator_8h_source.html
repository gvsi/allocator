160  pointer allocate(size_type n) {
161  if (n < 1) {
162  std::bad_alloc exception;
163  throw exception;
164  }
165 
166  int min_block_size = sizeof(value_type) + (2 * sizeof(int)); // 16
167  int alloc_req_bytes = n * sizeof(value_type); // 24
168  int min_req_bytes = n * sizeof(value_type) + min_block_size; // 40
169 
170  char *first_sent_ptr = a;
171  int *first_sent_int = reinterpret_cast<int *>(first_sent_ptr);
172  char *second_sent_ptr = first_sent_ptr + abs(*first_sent_int) + 4;
173 
174  // Check that the values in the two sentinels are the same
175  assert(*reinterpret_cast<int *>(first_sent_ptr) ==
176  *reinterpret_cast<int *>(second_sent_ptr));
177 
178  while (first_sent_ptr < a + N - 4) {
179  int *p = reinterpret_cast<int *>(first_sent_ptr);
180  int *q = reinterpret_cast<int *>(second_sent_ptr);
181  assert(*p == *q);
182 
183  if (*p == alloc_req_bytes) { // user gets exactly what was requested
184  *p = *p * -1;
185  *q = *q * -1;
186 
187  assert(valid());
188  return reinterpret_cast<pointer>(p + 1);
189  } else if (*p > alloc_req_bytes &&
190  *p < min_req_bytes) { // user gets more than what was requested
191 
192  *p = *p * -1;
193  *q = *q * -1;
194 
195  assert(valid());
196  return reinterpret_cast<pointer>(p + 1);
197  } else if (*p >= min_req_bytes) { // user gets exactly what was requested
198 
199  int remaining_free_bytes =
200  *p - alloc_req_bytes - 8; // the bytes that will be free on the
201  // adjacent block on the right (value
202  // will be displayed on the new sentinels)
203  *p = alloc_req_bytes * -1;
204  *reinterpret_cast<int *>(first_sent_ptr + 4 + alloc_req_bytes) =
205  alloc_req_bytes * -1; // sets new sentinel
206 
207  *reinterpret_cast<int *>(first_sent_ptr + 4 + alloc_req_bytes + 4) =
208  remaining_free_bytes;
209  *q = remaining_free_bytes;
210 
211  assert(valid());
212  return reinterpret_cast<pointer>(p + 1);
213  }
214 
215  // increment sentinels
216  first_sent_ptr = second_sent_ptr + 4;
217  second_sent_ptr = first_sent_ptr + abs(*(int *)(first_sent_ptr)) + 4;
218  }

249  void deallocate(pointer p, size_type) {
250  if (p == nullptr) {
251  std::invalid_argument exception("Invalid pointer");
252  throw exception;

255  assert(valid());
256 
257  int *first_sent_ptr = reinterpret_cast<int *>(p) - 1;
258  char *first_sent_char_ptr = reinterpret_cast<char *>(first_sent_ptr);
259  int *second_sent_ptr =
260  reinterpret_cast<int *>(first_sent_char_ptr + abs(*first_sent_ptr) + 4);
261 
262  if (*first_sent_ptr != *second_sent_ptr) {
263  std::cout << *first_sent_ptr << " " << *second_sent_ptr << std::endl;
264  }
265  assert(*first_sent_ptr == *second_sent_ptr);
266 
267  // Check whether there are adjancent blocks
268  bool is_first_block = first_sent_ptr == reinterpret_cast<int *>(a);
269  bool is_last_block = second_sent_ptr == reinterpret_cast<int *>(a + N) - 1;
270 
271  // Keeps track of all the bytes that we free after potential coalescing
272  int freed_bytes = abs(*first_sent_ptr);
273 
274  if (!is_first_block) {
275  int *prev_ptr = first_sent_ptr - 1;
276  if (*prev_ptr > 0) { // we can coalesce with previous block
277  // Increment freed bytes counter by size of adjacent free block
278  freed_bytes += *prev_ptr + 8;
279 
280  // Shift lower bound sentinel to previous block
281  first_sent_ptr = reinterpret_cast<int *>(
282  reinterpret_cast<char *>(prev_ptr) - *prev_ptr - 4);
283  assert(*first_sent_ptr > 0);
284  assert(*first_sent_ptr == *prev_ptr);
285  }
286  }
287 
288  if (!is_last_block) {
289  int *next_ptr = second_sent_ptr + 1;
290  if (*next_ptr > 0) { // we can coalesce with next block
291  // Increment freed bytes counter by size of adjacent free block
292  freed_bytes += *next_ptr + 8;
293 
294  // Shift upper bound sentinel to next block
295  second_sent_ptr = reinterpret_cast<int *>(
296  reinterpret_cast<char *>(next_ptr) + *next_ptr + 4);
297  assert(*second_sent_ptr > 0);
298  assert(*second_sent_ptr == *next_ptr);
299  }
300  }
301 
302  // Set the values of the resized sentinels
303  *first_sent_ptr = freed_bytes;
304  *second_sent_ptr = freed_bytes;
